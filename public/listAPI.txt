/alumni = mengembalikan 1 alumni(tahun lulus) yang tahunnya paling baru
/ekstra = mengembalikan 1 konten(judul konten) yang kelompoknya ekstra dan paling baru
/fasilitas = mengembalikan 1 konten(judul konten) yang kelompoknya fasilitas dan paling baru
/terkait = mengembalikan semua web terkait (nama web, link)
/pengumuman/home = mengembalikan 4 pengumuman pertama (judul,tanggal)
/berita/home = mengembalikan 2 berita pertama (judul,foto,kategori,tanggal,isi sebagian)
/testimoni = mengembalikan semua testimoni (foto,nama,isi,jurusan,univ)
/prestasi = mengembalikan 4 prestasi pertama (namalomba,juara,tingkat)

/ekstra/{namaekstra} = mengembalikan 1 ekstra (judul,isi) sesuai nama ekstra
/fasilitas/{namafasilitas} = mengembalikan 1 fasilitas (judul,isi) sesuai nama
/pengumuman/{judul} = mengembalikan 1 pengumuman (judul,isi) sesuai judul
/berita/{judul} = mengembalikan 1 berita (judul,isi,foto sampul) sesuai judul
/listalumni = mengembalikan semua alumni (tahun lulus)
/listfasilitas = mengembalikan semua konten(judul konten) yang kelompoknya fasilitas
/listekstra = mengembalikan semua konten(judul konten) yang kelompoknya ekstra
